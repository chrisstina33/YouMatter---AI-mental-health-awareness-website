def find_outliers_iqr(data, feature_column):
    """
    Find outliers in a given feature using the IQR method.

    Parameters:
    - data: pandas DataFrame
    - feature_column: str, the name of the column containing the feature

    Returns:
    - outliers: pandas DataFrame containing outliers
    """

    # Select the specified feature column
    feature_values = data[feature_column]

    # Calculate the first and third quartiles (Q1 and Q3)
    Q1 = feature_values.quantile(0.25)
    Q3 = feature_values.quantile(0.75)

    # Calculate the IQR (Interquartile Range)
    IQR = Q3 - Q1

    # Define the lower and upper bounds for outliers
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR

    # Find outliers
    outliers = data[(feature_values < lower_bound) | (feature_values > upper_bound)]
    
    print(f"Outliers ratio is {round(len(outliers)/len(data), 2)}")
    print(f"Low bound is {lower_bound}")
    print(f"Upper bound is {upper_bound}")
    
    return outliers

def regex_processes(text):
    # Remove hashtags using regular expression
    hashtag_pattern = r'#\w+'
    text_without_hashtags = re.sub(hashtag_pattern, '', text)

    # Remove sentences in quotation marks
    quote_pattern_1 = r'"[^"]*"'
    text_without_quotes = re.sub(quote_pattern_1, '', text_without_hashtags)
    
    # Remove sentences in quotation marks
    quote_pattern_2 = r'"(^")*"'
    text_without_quotes = re.sub(quote_pattern_2, '', text_without_quotes)

    # Remove emails using regular expression
    email_pattern = r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'
    text_without_emails = re.sub(email_pattern, '', text_without_quotes)

    # Remove links (URLs) using regular expression
    link_pattern = r'http[s]?://(?:[a-zA-Z]|[0-9]|[$-_@.&+]|[!*\\(\\),]|(?:%[0-9a-fA-F][0-9a-fA-F]))+'
    text_without_links = re.sub(link_pattern, '', text_without_emails)

    # Remove newline characters
    text_without_newlines = text_without_links.replace('\n', ' ')
    text_without_newlines = text_without_newlines.replace('\r', ' ')
    text_without_newlines = text_without_newlines.replace('\t', ' ')
    text_without_newlines = text_without_newlines.replace('/n', ' ')
    text_without_newlines = text_without_newlines.replace('/r', ' ')
    text_without_newlines = text_without_newlines.replace('/t', ' ')
    
    return text_without_newlines
def clean_puncts(text):

    # Remove punctuation using string.punctuation
    translator = str.maketrans("", "", string.punctuation)
    text_without_punct = text.translate(translator)

    return text_without_punct
def remove_stopwords(tokens):

    # Remove stopwords
    stop_words = set(stopwords.words('english'))
    removed_stopwords_data = [word for word in tokens if word.lower() not in stop_words]

    return removed_stopwords_data

def filter_non_english(tokens):
    english_tokens = [word for word in tokens if word.lower() in english_words]
    return english_tokens
def nlp_preprocessing(text, last_type="lemma"):
    """
    type can be stem or lemma
    """
    
    text = clean_emojis(text)
    text = regex_processes(text)
    text = clean_puncts(text)
    tokens = tokenize(text)
    tokens = remove_stopwords(tokens)
    
    if last_type =='stem':
        tokens = stemming(tokens)
    elif last_type =='lemma':
        tokens = lemmazation(tokens)
    else:
        raise ValueError("last_type parameter has been assigned wrong!. It can be stem or lemma.")

    tokens = filter_non_english(tokens)
    
    return tokens
